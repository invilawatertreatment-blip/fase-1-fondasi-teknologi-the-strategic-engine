// GANTI DENGAN ID SPREADSHEET BAPAK (SUDAH SAYA MASUKKAN) const SHEET_ID = '1PbmF4qevWS2G4DU0hkbZrAGouz37smFSs8BI30a_PNI'; function doGet(e) { const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet(); // 1. MODE BACA DATA (KHUSUS UNTUK PUSAT / HQ) if (e.parameter.mode === 'read') { const data = sheet.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // 2. MODE TULIS DATA (RADAR DARI SALES & KLIEN) const timestamp = new Date().toLocaleString('id-ID', { timeZone: 'Asia/Jakarta' }); const company = e.parameter.company || "-"; const address = e.parameter.address || "-"; const nama = e.parameter.nama || "-"; const phone = e.parameter.phone || "-"; const role = e.parameter.role || "-"; const action = e.parameter.action || "-"; // Format Kolom: A=Waktu, B=Perusahaan, C=Alamat, D=Kontak, E=No HP, F=Jabatan, G=Aktivitas sheet.appendRow([timestamp, company, address, nama, phone, role, action]); return ContentService.createTextOutput(JSON.stringify({"status": "sukses"})) .setMimeType(ContentService.MimeType.JSON); }
